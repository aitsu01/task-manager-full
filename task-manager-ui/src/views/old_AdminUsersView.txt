<script setup>
import { ref, onMounted } from "vue"
import api from "../services/api"

const users = ref([])
const loading = ref(true)
const error = ref(null)

const roles = ref([
  { id: 2, name: "User" },
  { id: 3, name: "Manager" }
])

const fetchUsers = async () => {
  try {
    const response = await api.get("/admin/users/pending")
    users.value = response.data
  } catch (err) {
    error.value = "Errore caricamento utenti"
  } finally {
    loading.value = false
  }
}

const approveUser = async (userId, roleId) => {
  try {
    await api.patch(`/admin/users/${userId}/approve`, {
      role_id: roleId
    })

    users.value = users.value.filter(u => u.id !== userId)
  } catch (err) {
    alert("Errore approvazione")
  }
}

onMounted(fetchUsers)
</script>

<template>
  <div class="min-h-screen bg-gray-100 p-10">

    <h1 class="text-3xl font-bold mb-8">Admin Panel</h1>

    <div v-if="loading">Caricamento...</div>
    <div v-if="error" class="text-red-500">{{ error }}</div>

    <div v-if="users.length === 0 && !loading"
         class="bg-white p-6 rounded-xl shadow">
      Nessun utente in attesa 🎉
    </div>

    <div v-for="user in users"
         :key="user.id"
         class="bg-white p-6 rounded-xl shadow mb-4 flex justify-between items-center">

      <div>
        <p class="font-semibold">{{ user.name }}</p>
        <p class="text-gray-500 text-sm">{{ user.email }}</p>
      </div>

      <div class="flex gap-3 items-center">
        <select v-model="user.selectedRole"
                class="border rounded px-2 py-1">
          <option disabled value="">Seleziona ruolo</option>
          <option v-for="role in roles"
                  :key="role.id"
                  :value="role.id">
            {{ role.name }}
          </option>
        </select>

        <button
          @click="approveUser(user.id, user.selectedRole)"
          class="bg-green-600 text-white px-4 py-2 rounded hover:bg-green-700">
          Approva
        </button>
      </div>

    </div>

  </div>
</template>
